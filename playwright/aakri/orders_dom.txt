                },
                // CRITICAL: These settings prevent JSON parsing
                dataType: 'binary',  // Tell jQuery not to parse as JSON
                xhrFields: { 
                    responseType: 'blob'  // Handle as blob for file download
                },
                // IMPORTANT: Don't let jQuery process the response
                processData: true,
                cache: false,
                
                success: function (blob, status, xhr) {
                    console.log('Download successful, blob size:', blob.size);
                    
                    // Check if we actually got a blob
                    if (!(blob instanceof Blob)) {
                        console.error('Response is not a blob:', blob);
                        alert('Download failed: Invalid response format');
                        return;
                    }
                    
                    // Check blob size
                    if (blob.size === 0) {
                        alert('The CSV file is empty. Please check your filters.');
                        return;
                    }

                    // Extract filename from Content-Disposition header
                    var disposition = xhr.getResponseHeader('Content-Disposition');
                    var filename = 'biowaste_orders.csv';
                    
                    if (disposition) {
                        // Handle both quoted and unquoted filenames
                        var matches = disposition.match(/filename[^;=\n]*=((['"]).*?\2|[^;\n]*)/);
                        if (matches && matches[1]) {
                            filename = matches[1].replace(/['"]/g, '');
                        }
                    }

                    // Create download link and trigger download
                    try {
                        var url = window.URL.createObjectURL(blob);
                        var link = document.createElement('a');
                        link.href = url;
                        link.download = filename;
                        
                        // Append to body, click, then remove
                        document.body.appendChild(link);
                        link.click();
                        document.body.removeChild(link);
                        
                        // Clean up the object URL
                        setTimeout(function() {
                            window.URL.revokeObjectURL(url);
                        }, 1000);
                        
                        console.log('File download initiated successfully');
                        
                    } catch (error) {
                        console.error('Error creating download link:', error);
                        alert('Failed to create download link. Please try again.');
                    }
                },
                
                error: function (xhr, status, error) {
                    console.error('Download failed:', {
                        status: status,
                        error: error,
                        responseStatus: xhr.status,
                        responseText: xhr.responseText
                    });
                    
                    // Try to parse error response if it's JSON
                    try {
                        var errorResponse = JSON.parse(xhr.responseText);
                        alert('Download failed: ' + (errorResponse.message || 'Server error'));
                    } catch (e) {
                        // If not JSON, show generic error
                        alert('Failed to download CSV file. Please try again.');
                    }
                },
                
                complete: function () {
                    // Re-enable button regardless of success/failure
                    $('#download-csv-btn').prop('disabled', false).text('Download CSV');
                }
            });
        });

                if (selectedFilter === 'phone_number') {
                    // Allow only numbers if option is phone number
                    let cleaned = inputValue.replace(/[^0-9+]/g, '');

                    // Format: keep +91 if present, max 10 digits
                    if (cleaned.startsWith('+91')) {
                        cleaned = '+91' + cleaned.substring(3).replace(/[^0-9]/g, '').substring(0, 10);
                    } else {
                        cleaned = cleaned.replace(/[^0-9]/g, '').substring(0, 10);
                    }

                    // Always set cleaned input
                    $(this).val(cleaned);

                    // Extract just the number part for validation
                    let numberPart = cleaned.startsWith('+91') ? cleaned.substring(3) : cleaned;

                    // Validate only when user has entered 10 digits
                    if (numberPart.length === 10) {
                        const startsWithValidDigit = /^[6-9]/.test(numberPart);
                        const isNotRepeatedDigits = !/^(\d)\1{9}$/.test(numberPart);

                        if (!(startsWithValidDigit && isNotRepeatedDigits)) {
                            // Invalid — clear or show warning
                            $(this).val('');

                } else if (selectedFilter === 'order_code') {
                    // Allow both letters and numbers, but no spaces id order code
